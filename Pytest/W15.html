Choose Your Testing Framework

pytest: Simple and Powerful

pytest is the most popular Python testing framework. It requires minimal boilerplate and has intuitive syntax.

Python - pytest example 
# test_risk_scorer.py
import pytest
from risk_scorer import calculate_risk_score

# Normal case: typical input
def test_calculate_risk_score_normal():
    finding = {
        "severity": "high",
        "cvss_score": 8.5,
        "exploitable": True
    }
    score = calculate_risk_score(finding)
    assert score == 85
    assert isinstance(score, int)

# Edge case: boundary values
def test_calculate_risk_score_max_cvss():
    finding = {
        "severity": "critical",
        "cvss_score": 10.0,
        "exploitable": True
    }
    score = calculate_risk_score(finding)
    assert score == 100  # Should cap at 100

# Invalid input: should raise exception
def test_calculate_risk_score_invalid_severity():
    finding = {
        "severity": "super-bad",  # Invalid!
        "cvss_score": 5.0
    }
    with pytest.raises(ValueError):
        calculate_risk_score(finding)

unittest: Python's Built-in Framework

unittest comes with Python‚Äîno installation needed. It uses a class-based approach inspired by Java's JUnit.

import unittest
from risk_scorer import calculate_risk_score

class TestRiskScorer(unittest.TestCase):
    
    def test_calculate_risk_score_normal(self):
        """Test with typical valid input."""
        finding = {
            "severity": "high",
            "cvss_score": 8.5,
            "exploitable": True
        }
        score = calculate_risk_score(finding)
        self.assertEqual(score, 85)
        self.assertIsInstance(score, int)
    
    def test_calculate_risk_score_max_cvss(self):
        """Edge case: maximum CVSS score."""
        finding = {
            "severity": "critical",

            "exploitable": True
        }
        score = calculate_risk_score(finding)
        self.assertEqual(score, 100)

        """Invalid input should raise ValueError."""
        finding = {
            "severity": "super-bad",

Designing High-Value Test Cases

Not all tests are equal. Focus on tests that catch real bugs. Every test should answer: "What could go wrong here?"

‚úÖ Normal Case

‚ñº

The "happy path"‚Äîtypical, expected input that your code should handle every day.

Examples:
  • Valid JSON with all required fields
  • Log entry with standard timestamp format
  • User input within expected ranges
  • File that exists and is readable

‚ö†Ô∏è Edge Case

‚ñº

Boundary conditions and unusual-but-valid inputs that often reveal bugs.

Examples:
  • Empty list or empty dictionary
  • Maximum/minimum allowed values
  • Single item in a list
  • Very long strings
  • Unicode characters
  • Timestamps at midnight or year boundaries

‚ùå Invalid Input

‚ñº

Bad data that should be rejected gracefully with clear error messages.

Examples:
  • Missing required fields
  • Wrong data types (string instead of int)
  • Values outside valid ranges
  • Malformed JSON
  • File that doesn't exist
  • None/null when not expected

JSON Input Validation




Never trust input. Validate everything: required fields, data types, and value ranges.


  






Python - Manual Validation 


def validate_finding(data: dict) -> dict:
    """
    Validate a security finding before processing.
    
    Raises:
        ValueError: If required fields are missing or invalid.
    """
    errors = []
    
    # Check required fields exist
    required_fields = ["id", "severity", "cvss_score"]
    for field in required_fields:
        if field not in data:
            errors.append(f"Missing required field: {{field}}")
    
    # Early return if missing required fields
    if errors:
        raise ValueError("Validation failed: " + "; ".join(errors))
    
    # Validate data types
    if not isinstance(data["id"], str):
        errors.append(f"'id' must be string, got {type(data['id']).__name__}")
    
    if not isinstance(data["cvss_score"], (int, float)):
        errors.append("'cvss_score' must be a number")
    
    # Validate value ranges
    valid_severities = ["low", "medium", "high", "critical"]
    if data.get("severity") not in valid_severities:
        errors.append(
            f"'severity' must be one of {valid_severities}, "
            f"got '{data.get('severity')}'"
        )
    
    cvss = data.get("cvss_score", 0)
    if isinstance(cvss, (int, float)) and not (0 <= cvss <= 10):
        errors.append(f"'cvss_score' must be 0-10, got {{cvss}}")
    
    if errors:
        raise ValueError("Validation failed: " + "; ".join(errors))
    
    return data








For complex validation, consider a schema-based approach:




Python - Schema-Based Validation 


# Define your schema as a dict
FINDING_SCHEMA = {
    "required": ["id", "severity", "cvss_score"],
    "types": {
        "id": str,
        "severity": str,
        "cvss_score": (int, float),
        "description": str,
        "exploitable": bool,
    },
    "allowed_values": {
        "severity": ["low", "medium", "high", "critical"],
    },
    "ranges": {
        "cvss_score": (0, 10),
    }
}

def validate_against_schema(data: dict, schema: dict) -> list:
    """Validate data against a schema definition."""
    errors = []
    
    # Check required fields
    for field in schema.get("required", []):
        if field not in data:
            errors.append(f"Missing required field: {{field}}")
    
    # Check types
    for field, expected_type in schema.get("types", {}).items():
        if field in data and not isinstance(data[field], expected_type):
            errors.append(
                f"'{{field}}' must be {expected_type}, "
                f"got {type(data[field]).__name__}"
            )
    
    # Check allowed values
    for field, allowed in schema.get("allowed_values", {}).items():
        if field in data and data[field] not in allowed:
            errors.append(
                f"'{{field}}' must be one of {{allowed}}"
            )
    
    # Check ranges
    for field, (min_val, max_val) in schema.get("ranges", {}).items():
        if field in data:
            val = data[field]
            if isinstance(val, (int, float)):
                if not (min_val <= val <= max_val):
                    errors.append(
                        f"'{{field}}' must be between "
                        f"{min_val} and {max_val}"
                    )
    
    return errors

        print("Hint: Check that the file path is correct.")

Logging for Troubleshooting




Good logs help you (and others) understand what your tool did and why it failed.




Python - Logging Setup 


import logging
from datetime import datetime

def setup_logging(log_file: str = None, level: str = "INFO"):
    """
    Configure logging for the application.
    
    Args:
        log_file: Path to log file (optional, logs to stderr if None)
        level: Logging level (DEBUG, INFO, WARNING, ERROR)
    """
    # Create formatter
    formatter = logging.Formatter(
        fmt="%(asctime)s | %(levelname)-8s | %(name)s | %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S"
    )
    
    # Get root logger
    root_logger = logging.getLogger()
    root_logger.setLevel(getattr(logging, level.upper()))
    
    # Console handler (always)
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(formatter)
    root_logger.addHandler(console_handler)
    
    # File handler (optional)
    if log_file:
        file_handler = logging.FileHandler(log_file)
        file_handler.setFormatter(formatter)
        root_logger.addHandler(file_handler)
    
    return root_logger

# Usage in your modules
logger = logging.getLogger(__name__)

def process_findings(findings: list) -> list:
    logger.info(f"Processing {len(findings)} findings")
    
    results = []
    for i, finding in enumerate(findings):
        logger.debug(f"Processing finding {i+1}: {finding.get('id')}")
        
        try:
            result = analyze_finding(finding)
            results.append(result)
        except Exception as e:
            logger.warning(
                f"Failed to process finding {finding.get('id')}: {{e}}"
            )
    
    logger.info(f"Successfully processed {len(results)}/{len(findings)} findings")
    return results
