import shutil from pathlib import Path source = Path("downloads/file.pdf") dest = Path("documents/file.pdf") # Move file shutil.move(str(source), str(dest)) # Copy instead shutil.copy2(str(source), str(dest))

üéØ Extension Handling

Handle edge cases: uppercase, no extension, multiple dots.

from pathlib import Path def get_extension(filename): path = Path(filename) ext = path.suffix.lower() # .PDF -> .pdf return ext[1:] if ext else "" # Examples get_extension("report.PDF") # "pdf" get_extension("README") # "" get_extension("file.tar.gz") # "gz"

Categorization Strategy

Design your mapping from file extensions to categories:

# Category mapping dictionary CATEGORIES = { "documents": ["pdf", "doc", "docx", "txt", "rtf", "odt"], "images": ["jpg", "jpeg", "png", "gif", "bmp", "svg", "webp"], "archives": ["zip", "tar", "gz", "rar", "7z"], "executables": ["exe", "msi", "bat", "sh", "app"], "videos": ["mp4", "avi", "mkv", "mov", "wmv"], "audio": ["mp3", "wav", "flac", "aac", "ogg"] } def categorize_file(filename): ext = get_extension(filename) for category, extensions in CATEGORIES.items(): if ext in extensions: return category return "other" # Default for unknown types
üí° Security Consideration: In a real-world scenario, be extra careful with executables. Consider quarantining suspicious files or adding hash verification. As cybersecurity professionals, always validate file types beyond just extensions!
