üìÅ File Input/Output: Reading and Writing Data

JSON: The Universal Data Format

JSON (JavaScript Object Notation) is the standard format for data exchange in cybersecurity tools. It's human-readable, widely supported, and perfect for configuration files and data storage.

Reading JSON Files

file_handler.py - Basic JSON Reading:
import json
import logging

def read_json_file(filepath):
    """
    Read and parse a JSON file.
    
    Args:
        filepath (str): Path to the JSON file
        
    Returns:
        dict: Parsed JSON data
        
    Raises:
        FileNotFoundError: If file doesn't exist
        json.JSONDecodeError: If file contains invalid JSON
    """
    try:
        logging.info(f"Reading JSON file: {{filepath}}")
        with open(filepath, 'r') as file:
            data = json.load(file)
        logging.info(f"Successfully loaded {len(data)} items")
        return data
    except FileNotFoundError:
        logging.error(f"File not found: {{filepath}}")
        raise
    except json.JSONDecodeError as e:
        logging.error(f"Invalid JSON in {{filepath}}: {{e}}")
        raise
    except Exception as e:
        logging.error(f"Unexpected error reading {{filepath}}: {{e}}")
        raise

Writing JSON Files

file_handler.py - Basic JSON Writing:
def write_json_file(filepath, data):
    """
    Write data to a JSON file.
    
    Args:
        filepath (str): Path to write the JSON file
        data: Data to write (must be JSON-serializable)
        
    Returns:
        bool: True if successful
    """
    try:
        logging.info(f"Writing JSON file: {{filepath}}")
        with open(filepath, 'w') as file:
            json.dump(data, file, indent=4)
        logging.info(f"Successfully wrote {{filepath}}")
        return True
    except TypeError as e:
        logging.error(f"Data not JSON-serializable: {{e}}")
        raise
    except IOError as e:
        logging.error(f"Error writing file {{filepath}}: {{e}}")
        raise

processor.py - Analyzing Security Events:
from datetime import datetime, timedelta
import logging

def filter_recent_events(events, hours=24):
    """
    Filter events that occurred within the last N hours.
    
    Args:
        events (list): List of event dictionaries
        hours (int): Number of hours to look back
        
    Returns:
        list: Filtered events
    """
    cutoff_time = datetime.now() - timedelta(hours=hours)
    recent_events = []
    
    for event in events:
        try:
            event_time = datetime.fromisoformat(event['timestamp'].replace('Z', '+00:00'))
            if event_time >= cutoff_time:
                recent_events.append(event)
        except (KeyError, ValueError) as e:
            logging.warning(f"Skipping malformed event: {{e}}")
            continue
    
    logging.info(f"Found {len(recent_events)} recent events out of {len(events)} total")
    return recent_events


def calculate_risk_score(event):
    """
    Calculate a risk score for a security event.
    
    Args:
        event (dict): Security event data
        
    Returns:
        int: Risk score (0-100)
    """
    score = 0
    
    # Base score by event type
    risk_weights = {
        'login_failure': 20,
        'unauthorized_access': 80,
        'malware_detected': 90,
        'config_change': 30,
        'port_scan': 40
    }
    
    score += risk_weights.get(event.get('event_type', ''), 10)
    
    # Increase score for high severity
    if event.get('severity') == 'critical':
        score += 30
    elif event.get('severity') == 'high':
        score += 20
    
    # Cap at 100
    return min(score, 100)


def aggregate_by_user(events):
    """
    Group events by user and count occurrences.
    
    Args:
        events (list): List of event dictionaries
        
    Returns:
        dict: User -> event count mapping
    """
    user_counts = {}
    
    for event in events:
        user = event.get('user', 'unknown')
        user_counts[user] = user_counts.get(user, 0) + 1
    
    # Sort by count (descending)
    sorted_users = dict(sorted(user_counts.items(), 
                              key=lambda x: x[1], 
                              reverse=True))
    
    logging.info(f"Aggregated events for {len(sorted_users)} users")
    return sorted_users

üéØ Object-Oriented Programming: Classes with Purpose

Why Use Classes?

Classes help you organize related data and functions together. Think of a class as a blueprint for creating objects that have both properties (attributes) and behaviors (methods).

When to Use a Class

‚úÖ Good Reasons

  • Representing a real entity (Host, User, Event)
  • Grouping related data and operations
  • Need multiple instances with same structure
  • Want to encapsulate complex state

‚ùå Avoid Classes For

  • Simple utility functions
  • One-off operations
  • When a dictionary would suffice
  • Over-engineering simple problems

Example: SecurityEvent Class

classes.py - A Practical Security Class:
from datetime import datetime
import logging

class SecurityEvent:
    """
    Represents a security event with analysis capabilities.
    """
    
    # Class variable - risk weight lookup
    RISK_WEIGHTS = {
        'login_failure': 20,
        'unauthorized_access': 80,
        'malware_detected': 90,
        'config_change': 30,
        'port_scan': 40
    }
    
    def __init__(self, timestamp, event_type, user, ip, severity='low'):
        """
        Initialize a SecurityEvent.
        
        Args:
            timestamp (str): ISO format timestamp
            event_type (str): Type of security event
            user (str): Username associated with event
            ip (str): Source IP address
            severity (str): Event severity level
        """
        self.timestamp = self._parse_timestamp(timestamp)
        self.event_type = event_type
        self.user = user
        self.ip = ip
        self.severity = severity
        self.risk_score = self._calculate_risk()
        
        logging.debug(f"Created SecurityEvent: {event_type} by {{user}}")
    
    def _parse_timestamp(self, timestamp_str):
        """Parse ISO format timestamp string."""
        try:
            return datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
        except ValueError:
            logging.warning(f"Invalid timestamp: {timestamp_str}, using current time")
            return datetime.now()
    
    def _calculate_risk(self):
        """Calculate risk score for this event."""
        base_score = self.RISK_WEIGHTS.get(self.event_type, 10)
        
        # Severity multiplier
        severity_bonus = {
            'low': 0,
            'medium': 10,
            'high': 20,
            'critical': 30
        }
        
        total_score = base_score + severity_bonus.get(self.severity, 0)
        return min(total_score, 100)
    
    def is_high_risk(self):
        """Check if this event is high risk (score > 50)."""
        return self.risk_score > 50
    
    def age_hours(self):
        """Get age of event in hours."""
        age = datetime.now() - self.timestamp
        return age.total_seconds() / 3600
    
    def to_dict(self):
        """Convert event to dictionary for JSON serialization."""
        return {
            'timestamp': self.timestamp.isoformat(),
            'event_type': self.event_type,
            'user': self.user,
            'ip': self.ip,
            'severity': self.severity,
            'risk_score': self.risk_score
        }
    
    def __repr__(self):
        """String representation for debugging."""
        return (f"SecurityEvent(type={self.event_type}, user={self.user}, "
                f"risk={self.risk_score})")
    
    def __str__(self):
        """User-friendly string representation."""
        return (f"[{self.timestamp}] {self.event_type.upper()} by {self.user} "
                f"from {self.ip} (Risk: {self.risk_score})")

üìù Logging and Error Handling: Know What's Happening

Why Logging Matters

Logging helps you understand what your program is doing, diagnose problems, and track security events. It's like having a flight recorder for your code.

Python Logging Basics

Setting Up Logging in main.py:
import logging
from datetime import datetime

def setup_logging(log_level='INFO'):
    """
    Configure logging for the application.
    
    Args:
        log_level (str): Logging level (DEBUG, INFO, WARNING, ERROR, CRITICAL)
    """
    # Create logs directory if it doesn't exist
    import os
    os.makedirs('logs', exist_ok=True)
    
    # Generate log filename with timestamp
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    log_file = f'logs/app_{{timestamp}}.log'
    
    # Configure logging format
    log_format = '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    
    # Set up logging to both file and console
    logging.basicConfig(
        level=getattr(logging, log_level.upper()),
        format=log_format,
        handlers=[
            logging.FileHandler(log_file),
            logging.StreamHandler()  # Also print to console
        ]
    )
    
    logging.info(f"Logging initialized. Log file: {log_file}")
    return log_file

Error Handling Pattern:
def process_file(filepath):
    """
    Process a file with comprehensive error handling.
    
    Args:
        filepath (str): Path to file
        
    Returns:
        dict: Processed data or None if error
    """
    try:
        # Try to open and process the file
        logging.info(f"Processing file: {{filepath}}")
        
        with open(filepath, 'r') as f:
            data = json.load(f)
        
        # Validate required fields
        required_fields = ['hosts', 'timestamp']
        for field in required_fields:
            if field not in data:
                raise ValueError(f"Missing required field: {{field}}")
        
        # Process the data
        results = analyze_data(data)
        logging.info(f"Successfully processed {len(results)} items")
        return results
        
    except FileNotFoundError:
        logging.error(f"File not found: {{filepath}}")
        print(f"ERROR: Cannot find file '{{filepath}}'")
        return None
        
    except json.JSONDecodeError as e:
        logging.error(f"Invalid JSON in {{filepath}}: {{e}}")
        print(f"ERROR: File '{{filepath}}' contains invalid JSON")
        return None
        
    except ValueError as e:
        logging.error(f"Validation error: {{e}}")
        print(f"ERROR: {{e}}")
        return None
        
    except Exception as e:
        # Catch-all for unexpected errors
        logging.error(f"Unexpected error processing {{filepath}}: {{e}}", exc_info=True)
        print(f"ERROR: An unexpected error occurred: {{e}}")
        return None

üíª Command-Line Interface: Making Your Tool Usable

Why CLI Matters

A good CLI makes your tool professional and easy to use. Users should be able to run your tool with clear arguments and get helpful feedback.

Using argparse for CLI

main.py - Complete CLI Implementation:
import argparse
import logging
import sys
from file_handler import read_json_file, write_json_file
from processor import process_data

def parse_arguments():
    """
    Parse command-line arguments.
    
    Returns:
        argparse.Namespace: Parsed arguments
    """
    parser = argparse.ArgumentParser(
        description='Security Event Analyzer - Process and analyze security logs',
        epilog='Example: python main.py -i data/events.json -o output/results.json'
    )
    
    # Required arguments
    parser.add_argument(
        '-i', '--input',
        required=True,
        help='Path to input JSON file'
    )
    
    parser.add_argument(
        '-o', '--output',
        required=True,
        help='Path to output JSON file'
    )
    
    # Optional arguments
    parser.add_argument(
        '-t', '--threshold',
        type=int,
        default=50,
        help='Risk score threshold (default: 50)'
    )
    
    parser.add_argument(
        '--hours',
        type=int,
        default=24,
        help='Hours to look back for recent events (default: 24)'
    )
    
    parser.add_argument(
        '-v', '--verbose',
        action='store_true',
        help='Enable verbose logging (DEBUG level)'
    )
    
    parser.add_argument(
        '--version',
        action='version',
        version='%(prog)s 1.0'
    )
    
    return parser.parse_args()


def main():
    """
    Main entry point for the application.
    """
    # Parse arguments
    args = parse_arguments()
    
    # Set up logging
    log_level = 'DEBUG' if args.verbose else 'INFO'
    setup_logging(log_level)
    
    logging.info("=" * 60)
    logging.info("Security Event Analyzer v1.0")
    logging.info("=" * 60)
    
    try:
        # Load input data
        logging.info(f"Loading input from: {args.input}")
        data = read_json_file(args.input)
        
        if data is None:
            logging.error("Failed to load input data")
            sys.exit(1)
        
        # Process data
        logging.info(f"Processing {len(data)} events...")
        results = process_data(
            data,
            threshold=args.threshold,
            hours=args.hours
        )
        
        # Write output
        logging.info(f"Writing results to: {args.output}")
        write_json_file(args.output, results)
        
        # Print summary
        print("\n" + "=" * 60)
        print("PROCESSING COMPLETE")
        print("=" * 60)
        print(f"Input file:     {args.input}")
        print(f"Output file:    {args.output}")
        print(f"Events processed: {len(data)}")
        print(f"High-risk events: {len(results['high_risk'])}")
        print("=" * 60)
        
        logging.info("Application completed successfully")
        return 0
        
    except Exception as e:
        logging.error(f"Application error: {{e}}", exc_info=True)
        print(f"\nFATAL ERROR: {{e}}")
        return 1


if __name__ == '__main__':
    sys.exit(main())

        >>> validate_ip_address("192.168.1.1")
        True
        >>> validate_ip_address("256.1.1.1")
        False
    """
    pattern = r'^(\d{1,3}\.){{3}}\d{1,3}$'
    
    if not re.match(pattern, ip_string):
        return False
    
    # Check each octet is 0-255
    octets = ip_string.split('.')
    return all(0 <= int(octet) <= 255 for octet in octets)
